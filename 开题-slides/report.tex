\begin{frame}
\frametitle{研究背景}
\begin{block}{挑战}
\begin{itemize}
  \item 救援任务复杂，不同类型智能体，多种任务，互相交错
  \pause
  \item 受限且不可靠的通信，只能感知局部信息
  \pause
  \item 高度动态变化，状态空间极大
  \pause
  \item 城市路况复杂，道路拥堵未知
\end{itemize}
\end{block}
\end{frame}

\begin{frame}
\frametitle{国内外研究现状}
\begin{itemize}
    \pause
  \item Bratman提出智能体信念(Brief)、愿望(Desire)、意图(Intention)的理性平衡观点，BDI模型成为研究智能体理性和推理机制的基础
  \pause
  \item Marie提出了部分全局规划PGP(Partial Global Planning)的方法，提供了一种多智能体系统的协调框架，适用于连续、动态的协调系统，允许各智能体进行动态协作
  \pause
    \item 杨东勇设计了一种基于协作协进化算法的分层式的智能体体系结构，通过协作协进化算法将多个神经网络的进化联系起来,使多个智能体产生协作
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{国内外研究现状}
\begin{itemize}
  \item \textbf{SEU-Jolly} 基于人工免疫的协作算法,基于粒子滤波的火势预测
  \item \textbf{ZJU-Base} AOE-Net带权与或树的动态任务协调
  \item \textbf{SOS} 基于Q学习的在线协作算法
\end{itemize}
\end{frame}


\section{Research Object}

\begin{frame}
\frametitle{研究目标}
构建一支强有力的救援队伍\\
在城市救援过程中，当处于复杂动态且人力资源、通信受限的环境\\
警察、救护、消防三类救援智能体进行有效协作、资源分配\\
提高整体救援效率\\
\end{frame}



\section{Research Content}
\begin{frame}
\frametitle{研究内容}
\begin{itemize}
  \item 城市救援环境中异构智能体的任务分配问题
  \item 救援智能体在复杂动态环境中动态路径规划问题
  \item 救援智能体信息融合机制
\end{itemize}
\end{frame}

\begin{frame}

\begin{itemize}
  \item (Why)Limit Info,难以形成对全局的认识

\end{block}
